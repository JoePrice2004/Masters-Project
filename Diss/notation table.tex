\section*{Notation and Symbol Definitions}
\
\begin{table}[hbt!] \label{Table3.1}
    \centering
    \begin{tabular}{>{$}c<{$} l}
        \toprule
        \text{Symbol} & \text{Definition} \\
        \midrule
        -----\text{Complex Analysis}----- & \\
        \midrule
        $i$& The imaginary number $(\sqrt{-1}=i)$ \\
        $$\Re(z)$$ & The real component of $z$ \\
        $$\Im(z)$$ & The imaginary component of $z$ \\
        $$z, s$$& Complex variables \\
        $$\sigma$$& Real part of $s$ i.e. $\sigma=\Re(s)$ \\
        $$\omega$$& Imaginary part of $s$ i.e. $\omega=\Im(s)$ \\
        $$\sigma_0$$& Abscissa of convergence \\
        $$\operatorname{Res}(f, z_0)$$& Residue of $f$ at $z_0$ \\
        $$\oint_\Gamma$$& Contour integral around closed contour $\Gamma$ \\
        \midrule
        -----\text{Contour and Lines}----- & \\
        \midrule
        $$Br$$& Bromwich line (vertical line $\Re(s)=\gamma$) \\
        $$\gamma$$& Real constant defining the Bromwich line: \\
        &$\gamma>\sigma_0$ \\
        $$C,\Gamma$$& General closed contours \\
        $$C_R$$& Large semicircular arc of radius $R$ \\
        $$\epsilon$$& Small circle around branch point \\
        \midrule
        -----\text{Transforms}----- & \\
        \midrule
        $$\mathcal{L}\{F(t)\}$$& Laplace transform of $F(t)$ \\
        $$f(s)$$ & Laplace transform of $F(t)$ \\
        $$\mathcal{L}^{-1}\{f(s)\}$$ & Inverse Laplace transform \\
        $$\mathcal{F}\{F(t)\}$$ & Fourier transform of $F(t)$ \\
        $$\hat{f}(\omega)$$ & Fourier transform of $f$ \\
        \midrule
        -----\text{Steepest descent}----- & \\
        \midrule
        $$I(k)$$ & Integral with large parameter $k$ \\
        $$\phi(z)$$ & Phase function \\
        $$f(z)$$ & Amplitude function \\
        $$z_0$$ & saddlepoint \\
        $$n$$ & Order of saddlepoint \\
        $$\alpha$$ & $\operatorname{arg}(\phi^n(z_0))$ phase of the n-th derivative \\
        $$\theta$$& steepest descent direction \\
        $$\beta$$ & Power law exponent of $f$ near $z_0$\\
        $$\Gamma(\cdot)$$& Gamma function\\
        \bottomrule
    \end{tabular}
    \caption{List of symbols and notations used in the analysis.}
\end{table}
